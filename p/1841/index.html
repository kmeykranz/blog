void Enemy::chase(Player *p1, Player *p2)
{
    int Xp1=p1->getX(),Yp1=p1->getY(),Xp2=p2->getX(),Yp2=p2->getY();
    int disP1=(x-Xp1)*(x-Xp1)+(y-Yp1)*(y-Yp1);
    int disP2=(x-Xp2)*(x-Xp2)+(y-Yp2)*(y-Yp2);
    if(disP1<disP2){
        if(x<Xp1){
            x+=speed;
            enemy->setPixmap(enemy_r);
        }
        else if(x>Xp1){
            x-=speed;
            enemy->setPixmap(enemy_l);
        }
        if(y<Yp1){
            y+=speed;
        }
        else if(y>Yp1){
            y-=speed;
        }
    }
    else{
        if(x<Xp2){
            x+=speed;
            enemy->setPixmap(enemy_r);
        }
        else if(x>Xp2){
            x-=speed;
            enemy->setPixmap(enemy_l);
        }
        if(y<Yp2){
            y+=speed;
        }
        else if(y>Yp2){
            y-=speed;
        }
    }
}
